PageTitle is a simple WebService that provides a RESTful api for retrieving page title from the URL, see example below
http://pagetitle.nodester.net/?uri=

Get Title



jQuery sample 1: 

$.ajax(
  {
    url:'http://pagetitle.duostack.net',
    data:{
      uri:'http://www.strx.it'
    },
    dataType:'jsonp',
    success:function(info){
      console.log('info:');
      console.dir(info);
    }
  }
);
jQuery sample 2: 

$.getJSON(
  'http://pagetitle.duostack.net?callback=?',
  {
    uri:'http://www.strx.it'
  },
  function(info){
    console.log('info:');
    console.dir(info);
  }
);
